.FUNC READ_LINE:
// Function to read a line of input into the buffer pointed to by r1
// r1 will point to the start of the buffer

PUSH r2
PUSH r3

ADDI r3, r0, 0x0A // Newline character

.FUNC DRAW_CHAR_TO_DISPLAY:
// Function to draw character in r1 to display at (r2, r3)
// r1: character to draw (ASCII)
// r2: X coordinate (pixels)
// r3: Y coordinate (pixels)
// Only use r1-r12, use stack for temps
PUSH r4
PUSH r5
PUSH r6
PUSH r7
PUSH r8
PUSH r9
PUSH r10
PUSH r11
PUSH r12
// r1: char, r2: x, r3: y
SUBI r1, r1, 35
LLI r4, BASE_FB_ADDR // font base
ADDI r4, r4, 2
MULI r5, r1, 8
ADD r4, r4, r5 // r4 = glyph ptr
LLI r5, 0x2000_0000 // fb base
LLI r6, 0 // row
DRAW_CHAR_ROW_LOOP:
LLI r12, 8
BGE r6, r12, END_DRAW_CHAR_LOOP
// For upside-down: use row = 7 - r6
LLI r12, 7
SUB r11, r12, r6
ADD r11, r4, r11
LB r7, r4, 0 // glyph row
LLI r8, 0 // col
DRAW_CHAR_COL_LOOP:
LLI r12, 8
BGE r8, r12, END_DRAW_CHAR_COL_LOOP
// For backwards: use col = r8 (if your font is stored MSB left)
SHR r10, r7, r8
ANDI r10, r10, 1
BEQ r10, r0, SKIP_PIXEL
// x = r2 + r8, y = r3 + r6
ADD r9, r2, r8 // x
ADD r10, r3, r6 // y
// pixel_index = y * 128 + x
MULI r11, r10, 128
ADD r11, r11, r9
// Save r11 (pixel_index) to stack
PUSH r11
// byte_index = pixel_index / 4
SHRI r12, r11, 2
ADD r12, r5, r12
// pixel_in_byte = pixel_index % 4
ANDI r11, r11, 3
LLI r9, 3
SUB r9, r9, r11
MULI r9, r9, 2 // bit_offset
// Read-modify-write framebuffer byte
LB r10, r12, 0
LLI r11, 0x3
SHL r11, r11, r9
NOT r11, r11, r0
AND r10, r10, r11
LLI r11, 0x3
SHL r11, r11, r9
OR r10, r10, r11
SB r10, r12, 0
// Restore r11
POP r11
SKIP_PIXEL:
ADDI r8, r8, 1
JMP DRAW_CHAR_COL_LOOP
END_DRAW_CHAR_COL_LOOP:
ADDI r4, r4, 1
ADDI r6, r6, 1
JMP DRAW_CHAR_ROW_LOOP
END_DRAW_CHAR_LOOP:
POP r12
POP r11
POP r10
POP r9
POP r8
POP r7
POP r6
POP r5
POP r4
RET
