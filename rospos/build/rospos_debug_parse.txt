.FUNC PRINT_STRING:
    // Function to print a character string pointed to by r1
    PUSH r2
    PUSH r3
    LLI r2, 0x1000_0000 // TTY address
PRINT_CHAR_LOOP:
    LB r3, r1, 0
    BEQ r3, r0, END_PRINT_CHAR_LOOP //error occurs around here
    SB r3, r2, 0
    ADDI r1, r1, 1
    JMP PRINT_CHAR_LOOP
END_PRINT_CHAR_LOOP:
    POP r3
    POP r2
    RET

.FUNC READ_CHAR:
    // Function to read a character from TTY into r1
    PUSH r2
    LLI r2, 0x1000_0000 // TTY address
    LB r1, r2, 0
    POP r2
    RET


.FUNC READ_LINE:
    // Function to read a line of input into the buffer pointed to by r1
    // r1 will point to the start of the buffer
    PUSH r1
    PUSH r2
    PUSH r3

READ_LINE_LOOP:
    CALL READ_CHAR
    ADDI r3, r0, 0x0A // Newline character
    BEQ r1, r3, END_READ_LINE_LOOP // End on newline
    SB r1, r2, 0
    ADDI r2, r2, 1
    JMP READ_LINE_LOOP

END_READ_LINE_LOOP:
    SB r0, r2, 0 // Null-terminate the string
    POP r3
    POP r2
    POP r1 // Restore original r1 (will point to the start of the buffer)
    RET

.FUNC DRAW_CHAR_TO_DISPLAY:
    // Function to draw character in r1 to display at (r2, r3)
    // r1: character to draw (ASCII)
    // r2: X coordinate (pixels)
    // r3: Y coordinate (pixels)
    // Only use r1-r12, use stack for temps
    PUSH r4
    PUSH r5
    PUSH r6
    PUSH r7
    PUSH r8
    PUSH r9
    PUSH r10
    PUSH r11
    PUSH r12
    // r1: char, r2: x, r3: y
    SUBI r1, r1, 34
    LLI r4, BASE_FB_ADDR // font base
    ADDI r4, r4, 2
    MULI r5, r1, 8
    ADD r4, r4, r5 // r4 = glyph ptr
    ADDI r4, r4, 1
    LLI r5, 0x2000_0000 // fb base
    LLI r6, 0 // row
DRAW_CHAR_ROW_LOOP:
    LLI r12, 8
    BGE r6, r12, END_DRAW_CHAR_LOOP
    // For upside-down: use row = 7 - r6
    LLI r12, 7
    SUB r11, r12, r6
    ADD r11, r4, r11
    LB r7, r4, 0 // glyph row
    LLI r8, 0 // col
DRAW_CHAR_COL_LOOP:
    LLI r12, 8
    BGE r8, r12, END_DRAW_CHAR_COL_LOOP
    LLI r12, 7
    SUB r9, r12, r8
    SHR r10, r7, r9
    ANDI r10, r10, 1
    BEQ r10, r0, SKIP_PIXEL
    // x = r2 + r8, y = r3 + r6
    ADD r9, r2, r8 // x
    ADD r10, r3, r6 // y
    // pixel_index = y * 256 + x
    MULI r11, r10, 256
    ADD r11, r11, r9
    // address = fb_base + pixel_index (1 byte per pixel)
    ADD r12, r5, r11
    // Store white pixel (00RRGGBB -> 0x3F)
    LLI r10, 0x3F
    SB r10, r12, 0
SKIP_PIXEL:
    ADDI r8, r8, 1
    JMP DRAW_CHAR_COL_LOOP
END_DRAW_CHAR_COL_LOOP:
    ADDI r4, r4, 1
    ADDI r6, r6, 1
    JMP DRAW_CHAR_ROW_LOOP
END_DRAW_CHAR_LOOP:
    POP r12
    POP r11
    POP r10
    POP r9
    POP r8
    POP r7
    POP r6
    POP r5
    POP r4
    RET

.FUNC CLEAR_CHAR_ON_DISPLAY:
    // Function to clear character at (r2, r3) by drawing black pixels
    // r2: X coordinate (pixels)
    // r3: Y coordinate (pixels)
    PUSH r4
    PUSH r5
    LLI r4, 0x2000_0000 // fb base
    LLI r5, 0 // row
CLEAR_CHAR_ROW_LOOP:
    LLI r12, 8
    BGE r5, r12, END_CLEAR_CHAR_LOOP
    LLI r6, 0 // col
CLEAR_CHAR_COL_LOOP:
    LLI r12, 8
    BGE r6, r12, END_CLEAR_CHAR_COL_LOOP
    // x = r2 + r6, y = r3 + r5
    ADD r9, r2, r6 // x
    ADD r10, r3, r5 // y
    // pixel_index = y * 256 + x
    MULI r11, r10, 256
    ADD r11, r11, r9
    // address = fb_base + pixel_index (1 byte per pixel)
    ADD r12, r4, r11
    // Store black pixel (0x00)
    SB r0, r12, 0
    ADDI r6, r6, 1
    JMP CLEAR_CHAR_COL_LOOP
END_CLEAR_CHAR_COL_LOOP:
    ADDI r5, r5, 1
    JMP CLEAR_CHAR_ROW_LOOP
END_CLEAR_CHAR_LOOP:
    POP r5
    POP r4
    RET

.FUNC DRAW_STRING_TO_DISPLAY:
    // Function to draw string pointed to by r1 to display at (r2, r3)
    // r1: pointer to string
    // r2: X coordinate (pixels)
    // r3: Y coordinate (pixels)
    PUSH r4
    PUSH r5
    LLI r4, 0 // char index
DRAW_STRING_LOOP:
    LB r5, r1, 0
    BEQ r5, r0, END_DRAW_STRING_LOOP // End on null terminator
    // Draw character
    CALL DRAW_CHAR_TO_DISPLAY
    // Advance string pointer and X coordinate
    ADDI r1, r1, 1
    ADDI r2, r2, 8 // Advance X by 8 pixels
    JMP DRAW_STRING_LOOP
END_DRAW_STRING_LOOP:
    POP r5
    POP r4
    RET

.FUNC CLEAR_DISPLAY:
    // Function to clear the display
    PUSH r1
    PUSH r2
    LLI r1, 0x2000_0000 // fb base
    LLI r2, 0x2010_0000 // fb end, 16KB framebuffer (256x256, 1 byte per pixel)
CLEAR_DISPLAY_LOOP:
    SB r0, r1, 0 // Write black pixel
    ADDI r1, r1, 1 // Move to next pixel
    BNE r1, r2, CLEAR_DISPLAY_LOOP // End when we reach the end of the framebuffer
    // After clearing the display, reset cursor to (0, 0)
    POP r2 
    POP r1 
    RET

//IDEA: Includes for files in binary/*
//binary_register.ros registered binaries declared in binary/*
//Then, has a function to, given a string buffer, check if it matches any registered binary
//If it does, call the corresponding loader function to load it into memory
//And run it
//This allows for easy inclusion of binary files into the VM
//Binaries are registered as a mapping from name to loader function
//Array of pairs of (name ptr, loader function ptr) stored in a special reserved block.

REGISTRY:
    .DATA BC_NAME // BC\0\0 - Name of binary, padded to a word
    .DATA BC      // Pointer to loader function for BC binary
    .SPACE 4 // "Pointer" to next entry set, but 0 because this is the end
    .SPACE 4
.FUNC LOAD_BINARY:
    // r1: pointer to null-terminated string with binary name
    // Returns: r1 = 0xFF if not found, otherwise return value depends on binary (e.g., it could return the binary's output in r1, or it could just run the binary and return when it's done)
    PUSH r2
    PUSH r3
    PUSH r4
    LW r4, r1, 0 // Load input name pointer
    LLI r2, REGISTRY
LOAD_BINARY_LOOP:
    LW r3, r2, 0 // Load name pointer
    BEQ r3, r0, LOAD_BINARY_NOT_FOUND // End of registry reached without match
    BEQ r3, r4, LOAD_BINARY_FOUND // Name matches
    ADDI r2, r2, 8 // Move to next registry entry (name ptr + loader ptr)
    JMP LOAD_BINARY_LOOP
LOAD_BINARY_FOUND:
    ADDI r2, r2, 4 // Move to loader function pointer
    LW r1, r2, 0 // Load loader function pointer into r1
    
    //For now, this is easier to hand-write than to encode in the assembler
    //As it doesn't really support the j-type instructions with 2-register operands.
    .DATA 0b0100_0001_1110_0001_0000_0000_0000_0000 // Encoding for JALR r14, r1, 0 
    // After binary finishes, we return here. It's up to the binary to decide what to do after it finishes (e.g., return to shell, halt, etc.)
    // If it does return, we'll assume that it either wrote its output to TTY, the display, memory, or in r1
    // And then we just return to the caller of LOAD_BINARY, which is likely the shell
    
    JMP LOAD_BINARY_CLEANUP
LOAD_BINARY_NOT_FOUND:
    LLI r1, 0xFF // Return 0xFF for not found
LOAD_BINARY_CLEANUP:
    POP r4
    POP r3
    POP r2
    RET

.FUNC SCREEN_TEST:
    PUSH r1
    PUSH r2
    PUSH r3
    PUSH r5
    PUSH r6
    
    LLI r1, 0x2000_0000 // Framebuffer base
    LLI r3, 0x0001_0000 // Loop counter (64K pixels)
    LLI r2, 0xBEEF     // Initial LFSR seed
    
SCREEN_TEST_LOOP:
    BEQ r3, r0, SCREEN_TEST_END
    
    SB r2, r1, 0       // Write pattern to framebuffer
    ADDI r1, r1, 1     // Move to next pixel
    
    // Update LFSR for next pattern value
    ADDI r5, r2, 0     // Copy current value to r5

    SHRI r2, r2, 1      // Shift right by 1
    BEQ r6, r0, NO_TAP
    XORI r2, r2, 0xB400 // Apply taps if LSB was 1
    
NO_TAP:
    SUBI r3, r3, 1     // Decrement counter
    JMP SCREEN_TEST_LOOP
    
SCREEN_TEST_END:
    POP r6
    POP r5
    POP r3
    POP r2
    POP r1
    RET
