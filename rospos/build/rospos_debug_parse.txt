.FUNC PRINT_STRING:
// Function to print a character string pointed to by r1

.FUNC READ_CHAR:
// Function to read a character from TTY into r1

.FUNC READ_LINE:
// Function to read a line of input into the buffer pointed to by r1
// r1 will point to the start of the buffer
PUSH r1
PUSH r2
PUSH r3

READ_LINE_LOOP:
CALL READ_CHAR
ADDI r3, r0, 0x0A // Newline character
BEQ r1, r3, END_READ_LINE_LOOP // End on newline

JMP READ_LINE_LOOP

POP r3
POP r2
POP r1 // Restore original r1 (will point to the start of the buffer)

.FUNC DRAW_CHAR_TO_DISPLAY:
// Function to draw character in r1 to display at (r2, r3)
// r1: character to draw
// r2: X coordinate
// r3: Y coordinate
PUSH r4
PUSH r5
PUSH r6
PUSH r7
PUSH r8

// Calculate font bitmap address
LLI r4, BASE_FB_ADDR
LLI r8, 0x20 // ASCII space character
SUB r5, r1, r8 // Offset from space character
LLI r8, 4 // Each character is 4 bytes
MUL r5, r5, r8 // No MULI
ADD r4, r4, r5 // r4 now has address of font bitmap
// Load font bitmap
LW r5, r4, 0
// Draw character pixel by pixel
LLI r6, 0 // Bit index
DRAW_CHAR_PIXEL_LOOP:
LLI r8, 32
BEQ r6, r8, END_DRAW_CHAR_PIXEL_LOOP // 32 bits per character
// Calculate pixel position
LLI r8, 8
DIV r7, r6, r8 // Row (0-3)
REM r9, r6, r8 // Column (0-7)
// Check if pixel is set
NEG r6, r0, r6
ADDI r6, r6, 31
SHR r1, r5, r6
ANDI r1, r1, 1
BEQ r1, r0, SKIP_PIXEL_DRAW
// Draw pixel to display
LLI r1, 0x2000_0000 // Display base address
LLI r8, 128 // Display width
MUL r2, r7, r8 // Y * width
ADD r1, r1, r2 // Base + Y offset
LLI r8, 4 // 2-bit grayscale, 4 pixels per byte
DIV r2, r1, r8 // Byte offset
REM r3, r1, r8 // Pixel within byte
LB r4, r2, 0 // Load byte
// Set pixel to white (0b11)
LLI r6, 0b11 // White pixel value
SHL r6, r6, r3 // Shift to correct position
OR r4, r4, r6 // Set pixel bits
SB r4, r2, 0 // Store byte back
SKIP_PIXEL_DRAW:
ADDI r6, r6, 1
JMP DRAW_CHAR_PIXEL_LOOP
END_DRAW_CHAR_PIXEL_LOOP:
POP r8
POP r7
POP r6
POP r5
POP r4
RET
