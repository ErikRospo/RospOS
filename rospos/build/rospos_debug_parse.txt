ELSE38:

ELSE40:

  JMP IF_END37
ELSE36:
IF_END37:
  ADDI r4, r3, 0    // assign prev
  ADDI r3, r3, 0    // assign current
  JMP WHILE34
WHILE_END35:
  // epilogue and return
  ADDI r1, r0, 0  // ensure r1=0
  POP r14
  RET
